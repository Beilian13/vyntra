${r.slice(0,2).toUpperCase()}

${r}

`; btn.addEventListener('click',()=>joinRoom(r)); container.appendChild(btn); }); } function joinRoom(room){ if(!connected) return alert("Not connected"); currentRoom=room; document.getElementById('messages').innerHTML=''; document.getElementById('roomTitle').textContent=room; document.getElementById('roomMeta').textContent="Loading recent messages..."; document.getElementById('roomUsersCount').textContent="0 users"; document.querySelectorAll('.room').forEach(r=>r.classList.remove('active')); const el=document.getElementById('room-'+room); if(el) el.classList.add('active'); ws.send(JSON.stringify({type:"join",room})); } function populateUsers(list){ const cont=document.getElementById('usersList'); cont.innerHTML=''; (list||[]).forEach(u=>{ const div=document.createElement('div'); div.className='user'; div.innerHTML=`
${u}
`; cont.appendChild(div); }); document.getElementById('roomUsersCount').textContent=`${list.length} users`; } function cacheAndRenderMessage(msg){ if(!msg||!msg.room) return; if(!messagesCache[msg.room]) messagesCache[msg.room]=[]; const idx=messagesCache[msg.room].findIndex(m=>m.id===msg.id); if(idx>=0) messagesCache[msg.room][idx]=msg; else messagesCache[msg.room].push(msg); if(msg.room===currentRoom) renderMessage(msg); } function pushSystemMessage(text){ const messages=document.getElementById('messages'); const el=document.createElement('div'); el.className='system'; el.textContent=text; messages.appendChild(el); messages.scrollTop=messages.scrollHeight; } function renderMessage(msg){ const container=document.getElementById('messages'); const existing=document.getElementById('msg-'+msg.id); if(existing){ const reactionsDiv=existing.querySelector('.reactions'); populateReactions(reactionsDiv,msg); return; } const wrapper=document.createElement('div'); wrapper.className='msg'+(msg.user===myUsernameLocal()?' me':''); wrapper.id='msg-'+msg.id; const meta=document.createElement('div'); meta.className='meta'; meta.textContent=`${msg.user} • ${new Date(parseInt(String(msg.id).split('_')[0])).toLocaleTimeString()}`; const text=document.createElement('div'); text.className='text'; text.textContent=msg.text; wrapper.appendChild(meta); wrapper.appendChild(text); const reactionsDiv=document.createElement('div'); reactionsDiv.className='reactions'; pickerBtn.title='Open emoji picker (circle)'; pickerBtn.addEventListener('click',()=>openEmojiPicker(msg.id)); pickerBtn.style.background='transparent'; reactionsDiv.appendChild(pickerBtn); wrapper.appendChild(reactionsDiv); container.appendChild(wrapper); container.scrollTop=container.scrollHeight; } function populateReactions(reactionsDiv,msg){ const picker=reactionsDiv.querySelector('button[title]'); reactionsDiv.innerHTML=''; let quick=[]; try{ quick=JSON.parse(reactionsDiv.dataset.quick||'["👍","❤️","😂"]'); }catch(e){ quick=["👍","❤️","😂"]; } quick.forEach(e=>{ const span=document.createElement('button'); span.className='reaction'; const count=(msg.reactions&&msg.reactions[e])?msg.reactions[e].length:0; span.innerHTML=`${e}${count>0?count:''}`; if(msg.reactions&&msg.reactions[e]&&msg.reactions[e].includes(myUsernameLocal())) span.classList.add('you'); span.className='reaction'; const count=msg.reactions[key].length; span.innerHTML=`${key}${count}`; if(msg.reactions[key].includes(myUsernameLocal())) span.classList.add('you'); span.onclick=()=>reactMessage(msg.id,key); pickerBtn.style.background='transparent'; reactionsDiv.appendChild(pickerBtn); reactionsDiv.dataset.quick=JSON.stringify(quick); } function sendLocalName(){ if(!connected||!myName) return; ws.send(JSON.stringify({type:"setname",name:myName}));
